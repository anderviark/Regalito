<script>

const container = document.getElementById('petalsContainer');

const counter = document.getElementById('counter');

const totalPetals = 20;

let left = totalPetals;

// Generación de pétalos complejos

for (let i = 0; i < totalPetals; i++) {

const angle = i * (360 / totalPetals);

const g = document.createElementNS("http://www.w3.org/2000/svg", "g");

g.setAttribute("class", "petal-group");

g.style.animationDelay = (i * 0.08) + 's';

const fill = (i % 2 === 0) ? "url(#grad1)" : "url(#grad2)";

const scale = 1.3 - (i * 0.01); // Variación sutil de tamaño

g.innerHTML = `<use href="#petalShape" transform="translate(250,350) rotate(${angle}) translate(0,-18) scale(${scale})" fill="${fill}" filter="url(#petalShadow)"/>`;

g.onclick = () => {

if (g.classList.contains('petal-fly')) return;

const rad = (angle - 90 + (Math.random()-0.5)*20) * (Math.PI / 180);

const power = 250 + Math.random() * 100;

g.style.setProperty('--fx', Math.cos(rad) * power + 'px');

g.style.setProperty('--fy', Math.sin(rad) * power - 100 + 'px'); // Caída parabólica

g.style.setProperty('--fr', ((Math.random()-0.5) * 500) + 'deg');

g.classList.add('petal-fly');

left--;

counter.innerText = left;

if (left === 0) {

setTimeout(() => document.getElementById('final').classList.add('show'), 1200);

}

};

container.appendChild(g);

}

// Sistema de Partículas (Brillitos de fondo)

const c = document.getElementById('particles'), ctx = c.getContext('2d');

let parts = [];

function resize() { c.width = window.innerWidth; c.height = window.innerHeight; }

window.onresize = resize; resize();

for(let i=0; i<60; i++) parts.push({

x: Math.random()*c.width, y: Math.random()*c.height,

s: Math.random()*2.5 + 0.5, vx: Math.random()-0.5, vy: Math.random()-0.5,

o: Math.random() * 0.7 + 0.1

});

function anim() {

ctx.clearRect(0,0,c.width,c.height);

parts.forEach(p => {

p.x+=p.vx * 0.6; p.y+=p.vy * 0.6;

if(p.x<0) p.x=c.width; if(p.x>c.width) p.x=0;

if(p.y<0) p.y=c.height; if(p.y>c.height) p.y=0;

ctx.globalAlpha = p.o;

ctx.fillStyle = (p.s > 2) ? "#fffbe0" : "#ffd8eb";

ctx.beginPath(); ctx.arc(p.x, p.y, p.s, 0, 7); ctx.fill();

});

ctx.globalAlpha = 1;

requestAnimationFrame(anim);

}

anim();

</script>
